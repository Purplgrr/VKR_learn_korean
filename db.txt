CREATE TABLE user (
    user_id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_login VARCHAR(50),
    user_password VARCHAR(50),
    user_name VARCHAR(50),
    lesson_id INTEGER,

    FOREIGN KEY (lesson_id) REFERENCES lesson(lesson_id) ON DELETE CASCADE
);

CREATE TABLE article_category (
    article_category_id INTEGER PRIMARY KEY AUTOINCREMENT,
    article_category_name VARCHAR(50)
);

CREATE TABLE article (
    article_id INTEGER PRIMARY KEY AUTOINCREMENT,
    article_name VARCHAR(50),
    article_content TEXT,
    article_category_id INTEGER,

    FOREIGN KEY (article_category_id) REFERENCES article_category(article_category_id) ON DELETE CASCADE
);

CREATE TABLE fav_article (
    article_id INTEGER,
    user_id INTEGER,

    PRIMARY KEY (article_id, user_id),
    FOREIGN KEY (article_id) REFERENCES article(article_id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES user(user_id) ON DELETE CASCADE
);
